# RAG Phase 2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add semantic and hybrid retrieval to the existing RAG module using an OpenAI-compatible embedding adapter and SQLite vector storage, while keeping the `knowledge-retrieval` node interface stable.

**Architecture:** Upgrade `SearchIndexAdapter` to a multi-channel seam (`SearchPlan` → `CandidateHit[]`). Add `EmbeddingAdapter`, `VectorIndexAdapter`, and `SegmentEmbedding` entity. Build keyword/semantic/hybrid adapters, dispatch by mode from a single `SEARCH_INDEX_ADAPTER`, and lazy-load embeddings during the first semantic retrieval.

**Tech Stack:** NestJS, TypeORM, better-sqlite3, OpenAI SDK (already installed), Node test runner.

## Global Constraints

- Embedding configuration is global via environment variables: `EMBEDDING_BASE_URL`, `EMBEDDING_API_KEY`, `EMBEDDING_MODEL`, `EMBEDDING_DIMENSIONS`.
- Vectors are stored as JSON arrays in SQLite; similarity is computed in memory.
- Hybrid mode runs keyword + semantic in parallel and deduplicates by `segmentId`, keeping the higher score. No score fusion.
- Existing `knowledge-retrieval` node input/output contract must remain unchanged.
- All hits after retrieval carry a `channel: "keyword" | "semantic"` field for trace/debugging.
- Lazy-load embeddings: generate and persist them on the first semantic search, not at index time.

```typescript
// backend/src/modules/rag/adapters/openai-embedding.adapter.ts
import { Injectable } from "@nestjs/common";
import OpenAI from "openai";
import { EmbeddingAdapter, EmbeddingError } from "./embedding.adapter";

@Injectable()
export class OpenAIEmbeddingAdapter implements EmbeddingAdapter {
  private readonly client?: OpenAI;
  private readonly model: string;
  private readonly dimensions?: number;

  constructor() {
    const apiKey = process.env.EMBEDDING_API_KEY;
    const baseURL =
      process.env.EMBEDDING_BASE_URL ?? "https://api.openai.com/v1";
    this.model = process.env.EMBEDDING_MODEL ?? "text-embedding-3-small";
    const dims = process.env.EMBEDDING_DIMENSIONS;
    this.dimensions = dims ? Number(dims) : undefined;

    if (apiKey) {
      this.client = new OpenAI({ apiKey, baseURL });
    }
  }

  async embed(texts: string[]): Promise<number[][]> {
    if (texts.length === 0) return [];
    if (!this.client) {
      throw new EmbeddingError("EMBEDDING_API_KEY is not configured");
    }

    try {
      const response = await this.client.embeddings.create({
        model: this.model,
        input: texts,
        ...(this.dimensions ? { dimensions: this.dimensions } : {}),
      });
      return response.data.map((item) => item.embedding);
    } catch (error) {
      throw new EmbeddingError(
        `Failed to generate embeddings with model ${this.model}`,
        error
      );
    }
  }
}
```

```typescript
// backend/src/modules/rag/adapters/sqlite-vector-index.adapter.ts
import { Injectable } from "@nestjs/common";
import { InjectRepository } from "@nestjs/typeorm";
import { In, Repository } from "typeorm";
import { SegmentEmbedding } from "../../database/entities/segment-embedding.entity";
import {
  VectorIndexAdapter,
  VectorSearchHit,
  VectorSearchQuery,
} from "./vector-index.adapter";

export function cosineSimilarity(a: number[], b: number[]): number {
  if (a.length !== b.length) return 0;
  let dot = 0;
  let normA = 0;
  let normB = 0;
  for (let i = 0; i < a.length; i++) {
    dot += a[i] * b[i];
    normA += a[i] * a[i];
    normB += b[i] * b[i];
  }
  if (normA === 0 || normB === 0) return 0;
  return Number((dot / (Math.sqrt(normA) * Math.sqrt(normB))).toFixed(6));
}

@Injectable()
export class SqliteVectorIndexAdapter implements VectorIndexAdapter {
  constructor(
    @InjectRepository(SegmentEmbedding)
    private readonly embeddingRepo: Repository<SegmentEmbedding>
  ) {}

  async upsert(embeddings: SegmentEmbedding[]): Promise<void> {
    if (embeddings.length === 0) return;
    await this.embeddingRepo.save(embeddings);
  }

  async deleteByDocument(documentId: string): Promise<void> {
    await this.embeddingRepo.delete({ documentId });
  }

  async deleteBySegmentIds(segmentIds: string[]): Promise<void> {
    if (segmentIds.length === 0) return;
    await this.embeddingRepo.delete({ segmentId: In(segmentIds) });
  }

  async search(input: VectorSearchQuery): Promise<VectorSearchHit[]> {
    if (input.datasetIds.length === 0) return [];

    const rows = await this.embeddingRepo.find({
      where: { segment: { datasetId: In(input.datasetIds) } },
      relations: { segment: true },
    });

    return rows
      .map((row) => ({
        segmentId: row.segmentId,
        score: cosineSimilarity(row.vector, input.queryVector),
      }))
      .filter((hit) => hit.score > 0)
      .sort((a, b) => b.score - a.score)
      .slice(0, input.topK);
  }
}
```

```typescript
// backend/src/modules/rag/adapters/sqlite-keyword-index.adapter.ts
import { Injectable } from "@nestjs/common";
import { InjectRepository } from "@nestjs/typeorm";
import { In, Repository } from "typeorm";
import { DocumentSegmentIndex } from "../../../database/entities/document-segment-index.entity";
import {
  CandidateHit,
  IndexedSegment,
  SearchIndexAdapter,
  SearchPlan,
} from "./search-index.adapter";

@Injectable()
export class SqliteKeywordIndexAdapter implements SearchIndexAdapter {
  constructor(
    @InjectRepository(DocumentSegmentIndex)
    private readonly indexRepo: Repository<DocumentSegmentIndex>
  ) {}

  async upsertSegments(segments: IndexedSegment[]): Promise<void> {
    if (segments.length === 0) return;

    await this.indexRepo.delete({
      segmentId: In(segments.map((segment) => segment.segmentId)),
    });
    await this.indexRepo.save(
      segments.map((segment) =>
        this.indexRepo.create({
          segmentId: segment.segmentId,
          datasetId: segment.datasetId,
          documentId: segment.documentId,
          content: segment.content,
        })
      )
    );
  }

  async deleteByDocument(documentId: string): Promise<void> {
    await this.indexRepo.delete({ documentId });
  }

  async search(plan: SearchPlan): Promise<CandidateHit[]> {
    if (plan.mode !== "keyword") return [];
    if (plan.datasetIds.length === 0) return [];

    const rows = await this.indexRepo.find({
      where: { datasetId: In(plan.datasetIds) },
    });

    const query = plan.query.trim().toLowerCase();
    const terms = Array.from(new Set(query.split(/\s+/g).filter(Boolean)));

    return rows
      .map((row) => ({
        segmentId: row.segmentId,
        score: this.scoreContent(row.content, query, terms),
        channel: "keyword" as const,
      }))
      .filter((row) => row.score > 0)
      .sort((a, b) => b.score - a.score)
      .slice(0, plan.topK);
  }

  private scoreContent(
    content: string,
    query: string,
    terms: string[]
  ): number {
    const normalized = content.toLowerCase();
    if (!normalized) return 0;

    let score = 0;
    if (query && normalized.includes(query)) {
      score += 1;
    }

    for (const term of terms) {
      if (!term) continue;
      const occurrences = normalized.split(term).length - 1;
      if (occurrences > 0) {
        score += Math.min(occurrences, 4) / Math.max(terms.length, 1);
      }
    }

    return Number(score.toFixed(4));
  }
}
```

```typescript
// backend/src/modules/rag/adapters/sqlite-semantic-search.adapter.ts
import { Injectable } from "@nestjs/common";
import { InjectRepository } from "@nestjs/typeorm";
import { In, Repository } from "typeorm";
import { DocumentSegment } from "../../../database/entities/document-segment.entity";
import { SegmentEmbedding } from "../../../database/entities/segment-embedding.entity";
import { EmbeddingAdapter } from "./embedding.adapter";
import {
  CandidateHit,
  SearchIndexAdapter,
  SearchPlan,
} from "./search-index.adapter";
import { VectorIndexAdapter } from "./vector-index.adapter";

const EMBEDDING_BATCH_SIZE = 64;

@Injectable()
export class SqliteSemanticSearchAdapter implements SearchIndexAdapter {
  constructor(
    @InjectRepository(DocumentSegment)
    private readonly segmentRepo: Repository<DocumentSegment>,
    @InjectRepository(SegmentEmbedding)
    private readonly embeddingRepo: Repository<SegmentEmbedding>,
    private readonly embeddingAdapter: EmbeddingAdapter,
    private readonly vectorIndex: VectorIndexAdapter
  ) {}

  async deleteByDocument(documentId: string): Promise<void> {
    await this.vectorIndex.deleteByDocument(documentId);
  }

  async search(plan: SearchPlan): Promise<CandidateHit[]> {
    if (plan.mode !== "semantic") return [];
    if (plan.datasetIds.length === 0) return [];

    const queryVectors = await this.embeddingAdapter.embed([plan.query]);
    const queryVector = queryVectors[0];
    if (!queryVector) return [];

    await this.ensureEmbeddings(plan.datasetIds);

    const hits = await this.vectorIndex.search({
      datasetIds: plan.datasetIds,
      queryVector,
      topK: plan.topK,
    });

    return hits.map((hit) => ({ ...hit, channel: "semantic" as const }));
  }

  private async ensureEmbeddings(datasetIds: string[]): Promise<void> {
    const segments = await this.segmentRepo.find({
      where: { datasetId: In(datasetIds) },
    });
    if (segments.length === 0) return;

    const existing = await this.embeddingRepo.find({
      where: { segmentId: In(segments.map((segment) => segment.id)) },
    });
    const existingBySegmentId = new Map(
      existing.map((embedding) => [embedding.segmentId, embedding])
    );

    const currentModel =
      process.env.EMBEDDING_MODEL ?? "text-embedding-3-small";
    const missing = segments.filter((segment) => {
      const embedding = existingBySegmentId.get(segment.id);
      return !embedding || embedding.model !== currentModel;
    });

    for (let i = 0; i < missing.length; i += EMBEDDING_BATCH_SIZE) {
      const batch = missing.slice(i, i + EMBEDDING_BATCH_SIZE);
      const vectors = await this.embeddingAdapter.embed(
        batch.map((segment) => segment.content)
      );

      const embeddings = batch.map((segment, index) =>
        this.embeddingRepo.create({
          segmentId: segment.id,
          documentId: segment.documentId,
          vector: vectors[index],
          model: currentModel,
        })
      );

      await this.vectorIndex.upsert(embeddings);
    }
  }
}
```

```typescript
// backend/src/modules/rag/adapters/sqlite-hybrid-search.adapter.ts
import { Injectable } from "@nestjs/common";
import {
  CandidateHit,
  SearchIndexAdapter,
  SearchPlan,
} from "./search-index.adapter";
import { SqliteKeywordIndexAdapter } from "./sqlite-keyword-index.adapter";
import { SqliteSemanticSearchAdapter } from "./sqlite-semantic-search.adapter";

@Injectable()
export class SqliteHybridSearchAdapter implements SearchIndexAdapter {
  constructor(
    private readonly keywordAdapter: SqliteKeywordIndexAdapter,
    private readonly semanticAdapter: SqliteSemanticSearchAdapter
  ) {}

  async deleteByDocument(documentId: string): Promise<void> {
    await this.keywordAdapter.deleteByDocument(documentId);
    await this.semanticAdapter.deleteByDocument(documentId);
  }

  async search(plan: SearchPlan): Promise<CandidateHit[]> {
    if (plan.mode !== "hybrid") return [];

    const [keywordHits, semanticHits] = await Promise.all([
      this.keywordAdapter.search({ ...plan, mode: "keyword" }),
      this.semanticAdapter.search({ ...plan, mode: "semantic" }),
    ]);

    const hitById = new Map<string, CandidateHit>();
    for (const hit of keywordHits) {
      hitById.set(hit.segmentId, hit);
    }
    for (const hit of semanticHits) {
      const existing = hitById.get(hit.segmentId);
      if (!existing || hit.score > existing.score) {
        hitById.set(hit.segmentId, hit);
      }
    }

    return Array.from(hitById.values())
      .sort((a, b) => b.score - a.score)
      .slice(0, plan.topK);
  }
}
```

```typescript
// backend/src/modules/rag/adapters/sqlite-search-index.adapter.ts
import { Injectable } from "@nestjs/common";
import {
  CandidateHit,
  SearchIndexAdapter,
  SearchPlan,
} from "./search-index.adapter";
import { SqliteHybridSearchAdapter } from "./sqlite-hybrid-search.adapter";
import { SqliteKeywordIndexAdapter } from "./sqlite-keyword-index.adapter";
import { SqliteSemanticSearchAdapter } from "./sqlite-semantic-search.adapter";

@Injectable()
export class SqliteSearchIndexAdapter implements SearchIndexAdapter {
  constructor(
    private readonly keywordAdapter: SqliteKeywordIndexAdapter,
    private readonly semanticAdapter: SqliteSemanticSearchAdapter,
    private readonly hybridAdapter: SqliteHybridSearchAdapter
  ) {}

  async deleteByDocument(documentId: string): Promise<void> {
    await this.keywordAdapter.deleteByDocument(documentId);
    await this.semanticAdapter.deleteByDocument(documentId);
  }

  async search(plan: SearchPlan): Promise<CandidateHit[]> {
    switch (plan.mode) {
      case "semantic":
        return this.semanticAdapter.search(plan);
      case "hybrid":
        return this.hybridAdapter.search(plan);
      case "keyword":
      default:
        return this.keywordAdapter.search(plan);
    }
  }
}
```
